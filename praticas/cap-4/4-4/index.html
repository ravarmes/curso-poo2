class Buffer {
    private Queue<Integer> fila = new LinkedList<>();
    private int capacidade = 5;
    
    public synchronized void produzir(int item) throws InterruptedException {
        // Espera se buffer cheio
        while (fila.size() == capacidade) {
            wait();  // Libera lock, espera
        }
        fila.add(item);
        notifyAll();  // Acorda consumidores
    }
    
    public synchronized int consumir() throws InterruptedException {
        // Espera se buffer vazio
        while (fila.isEmpty()) {
            wait();
        }
        int item = fila.poll();
        notifyAll();  // Acorda produtores
        return item;
    }
}
