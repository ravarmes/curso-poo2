Servidor Multi-cliente

public class ServidorChat {
    // Lista de todos os clientes conectados
    private static List<ClientHandler> clientes = 
        Collections.synchronizedList(new ArrayList<>());
    
    public static void main(String[] args) throws IOException {
        ServerSocket server = new ServerSocket(12345);
        System.out.println("Chat iniciado na porta 12345");
        
        while (true) {
            Socket socket = server.accept();
            
            // Cria handler e inicia thread
            ClientHandler handler = new ClientHandler(socket, clientes);
            clientes.add(handler);
            
            new Thread(handler).start();
        }
    }
    
    // Envia mensagem para todos os clientes
    public static void broadcast(String msg) {
        synchronized(clientes) {
            for (ClientHandler c : clientes) {
                c.enviar(msg);
            }
        }
    }
}

ClientHandler (Thread por Cliente)

class ClientHandler implements Runnable {
    private Socket socket;
    private PrintWriter out;
    private String nome;
    
    public ClientHandler(Socket socket) {
        this.socket = socket;
    }
    
    @Override
    public void run() {
        try {
            BufferedReader in = new BufferedReader(
                new InputStreamReader(socket.getInputStream()));
            out = new PrintWriter(socket.getOutputStream(), true);
            
            // Primeira mensagem = nome
            nome = in.readLine();
            ServidorChat.broadcast(nome + " entrou no chat!");
            
            String msg;
            while ((msg = in.readLine()) != null) {
                ServidorChat.broadcast(nome + ": " + msg);
            }
        } catch (IOException e) {
            // Cliente desconectou
        }
    }
    
    public void enviar(String msg) {
        out.println(msg);
    }
}
