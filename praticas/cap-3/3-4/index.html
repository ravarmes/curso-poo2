// 1. Model (Aluno.java)
public class Aluno {
    private int id;
    private String nome, email;
    // getters, setters, construtor...
}

// 2. DAO Interface
public interface AlunoDAO {
    void inserir(Aluno a);
    List<Aluno> listar();
}

// 3. Factory
public class DAOFactory {
    public static AlunoDAO createAlunoDAO() {
        return new AlunoDAOJDBC(getConnection());
    }
}

// 4. Controller
public class AlunoController {
    @FXML TextField txtNome;
    @FXML TableView<Aluno> tabela;
    
    private AlunoDAO dao = DAOFactory.createAlunoDAO();
    
    @FXML
    void salvar() {
        Aluno a = new Aluno(txtNome.getText());
        dao.inserir(a);
        tabela.setItems(FXCollections.observableList(dao.listar()));
    }
}
